def initialize(name, rating, meals_qty)
  @name = name
  @rating = rating
  @meals = Hash.new
  @meals["other"] = Meal.new("other", meals_qty)

    # File lib/restaurant.rb, line 29
def addMeal(meal)
  @meals[meal.name] = meal
  @meals["other"].remove_qty(meal.quantity)
end
